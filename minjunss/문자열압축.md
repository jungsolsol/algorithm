# 문자열 압축
```java
        // ArrayList<String> 생성
        // ArrayList<Integer> 생성
        // str 을 1개씩, 2개씩, ... 절반길이 까지 자르는 for 문
        // i개씩 자른 문자열을 ArrayList 에 넣음
        // int number = 1
        // ArrayList 의 index i, i+1이 같으면 i 인덱스값 ""로 변경 number++
        // i, i+1이 다르고 number >= 2 일때 i를 number+i로 변경
        // String result = String.join("",list)
        // ArrayList<Integer>에 result 길이 add
        // ArrayList<Integer> 최소값 리턴
       
       List<String> strList = new ArrayList<>();
        List<Integer> intList = new ArrayList<>();
        int strLen = str.length();
        int halfLen = str.length() / 2;
        for(int i=1; i<halfLen; i++) {
            int rest = strLen % i;
            for(int j=0; j<strLen-rest; j+=i) {
                strList.add(str.substring(j,j+i));
            }
            if(rest != 0) strList.add(str.substring(strLen-rest, strLen));

            for(int j=0; j<strList.size(); j++) {
                int number = 1;
                if(Objects.equals(strList.get(j), strList.get(j + 1))) {
                    strList.set(j,"");
                    number++;
                }
                if(!Objects.equals(strList.get(j), strList.get(j + 1)) && number >=2) {
                    strList.set(j,number+strList.get(j));
                }
            }
            String result = String.join("",strList);
            intList.add(result.length());
        }
        return Collections.min(intList);
    }
}
```
